Sticky stack kód

Minden kártya sticky top:0. Az előző scale + fade scroll-scrubbal. Canonical GSAP skeleton.

const cards = gsap.utils.toArray('.stack-card');

cards.forEach((card, i) => {
  if (i === cards.length - 1) return;
  // Pin each card at viewport top
  ScrollTrigger.create({
    trigger: card,
    start: "top top",
    endTrigger: cards[cards.length - 1],
    end: "top top",
    pin: true,
    pinSpacing: false,
  });
  // Scale + fade as next card arrives
  gsap.to(card, {
    scale: 0.9,
    opacity: 0.4,
    ease: "none",
    scrollTrigger: {
      trigger: cards[i + 1],
      start: "top bottom",
      end: "top top",
      scrub: true,
    },
  });
});
