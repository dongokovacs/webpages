Scrub szavanként – kód
// Szavak szétbontása
const words = el.textContent.split(' ');
el.innerHTML = words
  .map(w => `<span class="word">${w}</span>`)
  .join(' ');

// Scrub animáció
gsap.to('.word', {
  opacity: 1,
  stagger: 0.05,
  ease: 'none',
  scrollTrigger: {
    trigger: el,
    start: 'top 80%',
    end: 'bottom 30%',
    scrub: 1,
  }
});
