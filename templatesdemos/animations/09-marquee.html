/* A trükk: duplikált tartalom + max-content */
.mq-track {
  display: flex;
  width: max-content;       /* wrap nélkül egy sorban */
  animation: mq 20s linear infinite;
}
/* Pont -50% az animáció vége = seamless loop */
@keyframes mq {
  from { transform: translateX(0); }
  to   { transform: translateX(-50%); }
}
/* Hover pause */
.mq-wrap:hover .mq-track {
  animation-play-state: paused;
}
/* Fordított irány */
.mq-track.reverse { animation-name: mq-rev; }
@keyframes mq-rev {
  from { transform: translateX(-50%); }
  to   { transform: translateX(0); }
}
